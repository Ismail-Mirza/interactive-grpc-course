🔄 Exponential Backoff Reconnection

The most critical production pattern: when the stream drops, the dplane reconnects with exponential backoff to avoid thundering-herd problems when the cplane restarts.

pythongrpc-demo/dplane/reconnect.py
import time
import math
import logging
import random

logger = logging.getLogger(__name__)


class ExponentialBackoff:
    """
    Exponential backoff with jitter for reconnection.
    Pattern: wait = min(base * 2^attempt, max_delay) + random jitter
    """
    def __init__(
        self,
        base_delay: float = 1.0,
        max_delay: float = 60.0,
        multiplier: float = 2.0,
        jitter: float = 0.1,
    ):
        self.base_delay = base_delay
        self.max_delay = max_delay
        self.multiplier = multiplier
        self.jitter = jitter
        self._attempt = 0

    def next_delay(self) -> float:
        delay = min(
            self.base_delay * (self.multiplier ** self._attempt),
            self.max_delay
        )
        # Add ±10% jitter to prevent thundering herd
        jitter_amount = delay * self.jitter * (2 * random.random() - 1)
        self._attempt += 1
        return max(0, delay + jitter_amount)

    def reset(self):
        self._attempt = 0

    @property
    def attempt(self):
        return self._attempt


# Delays for attempts 0-7:
# 1.0s, 2.0s, 4.0s, 8.0s, 16.0s, 32.0s, 60.0s, 60.0s (capped)
pythongrpc-demo/dplane/client.py — full reconnection loop
import grpc
import threading
import time
import uuid
import logging
from reconnect import ExponentialBackoff

logger = logging.getLogger(__name__)


class DPlaneClient:
    def __init__(self, server_address, client_id, tenant_id, certs_dir=None):
        self.server_address = server_address
        self.client_id = client_id
        self.tenant_id = tenant_id
        self.certs_dir = certs_dir
        self.auth_token = None
        self.session_id = None
        self._running = False
        self._backoff = ExponentialBackoff(base_delay=1.0, max_delay=60.0)

    def _create_channel(self):
        if self.certs_dir:
            with open(f'{self.certs_dir}/ca.crt', 'rb') as f:
                root_ca = f.read()
            creds = grpc.ssl_channel_credentials(root_certificates=root_ca)
            return grpc.secure_channel(self.server_address, creds)
        return grpc.insecure_channel(self.server_address)

    def _metadata(self):
        meta = [('x-client-id', self.client_id)]
        if self.auth_token:
            meta.append(('authorization', f'Bearer {self.auth_token}'))
        return meta

    def _register(self, stub):
        """Register and get a JWT token."""
        resp = stub.RegisterClient(
            cplane_pb2.RegisterRequest(
                client_id=self.client_id,
                tenant_id=self.tenant_id,
                auth_token="bootstrap-secret",  # initial secret
                version="1.0.0",
            ),
            timeout=10.0,
        )
        if resp.success:
            self.session_id = resp.session_id
            self.auth_token = getattr(resp, 'jwt_token', None)
            logger.info(f"Registered: session={self.session_id}")
            return True
        return False

    def _stream_loop(self, stub):
        """Run the bidirectional stream with heartbeats."""
        import queue as q_module
        send_queue = q_module.Queue()
        heartbeat_event = threading.Event()

        def heartbeat_thread():
            while not heartbeat_event.wait(timeout=5.0):
                send_queue.put(cplane_pb2.ClientMessage(
                    message_id=str(uuid.uuid4()),
                    type=cplane_pb2.HEARTBEAT,
                    client_id=self.client_id,
                ))

        def message_generator():
            while True:
                try:
                    msg = send_queue.get(timeout=1.0)
                    if msg is None:
                        return
                    yield msg
                except Exception:
                    continue

        hb_thread = threading.Thread(target=heartbeat_thread, daemon=True)
        hb_thread.start()

        try:
            stream = stub.StreamMessages(
                message_generator(),
                metadata=self._metadata(),
            )
            for server_msg in stream:
                self._handle_message(server_msg, send_queue)
        finally:
            heartbeat_event.set()
            send_queue.put(None)

    def _handle_message(self, msg, send_queue):
        if msg.type == cplane_pb2.HEARTBEAT:
            logger.debug("Heartbeat ACK")
            return

        field = msg.WhichOneof('payload')
        if field == 'api_payload':
            api = msg.api_payload
            logger.info(f"Command: {api.action} {api.path}")
            # Execute and send callback
            send_queue.put(cplane_pb2.ClientMessage(
                message_id=str(uuid.uuid4()),
                type=cplane_pb2.RESPONSE,
                client_id=self.client_id,
                callback_payload=cplane_pb2.CallbackPayload(
                    request_id=msg.message_id,
                    result=f"executed {api.path}",
                    success=True,
                )
            ))

    def run(self):
        """Main loop: connect, register, stream, reconnect on failure."""
        self._running = True
        logger.info(f"dplane client starting: {self.client_id}")

        while self._running:
            channel = None
            try:
                channel = self._create_channel()
                stub = cplane_pb2_grpc.ControlPlaneStub(channel)

                # Register first
                if not self._register(stub):
                    raise RuntimeError("Registration failed")

                # Connection successful — reset backoff
                self._backoff.reset()
                logger.info("Connected to cplane — starting stream")

                # Start streaming (blocks until stream ends)
                self._stream_loop(stub)

                logger.info("Stream ended cleanly")

            except grpc.RpcError as e:
                code = e.code()
                logger.warning(f"gRPC error: {code} — {e.details()}")

                if code == grpc.StatusCode.UNAUTHENTICATED:
                    logger.error("Auth failed — check credentials")
                    time.sleep(30)  # longer wait for auth issues

                elif code in (grpc.StatusCode.UNAVAILABLE,
                              grpc.StatusCode.DEADLINE_EXCEEDED):
                    delay = self._backoff.next_delay()
                    logger.info(f"Server unavailable. Reconnect in {delay:.1f}s "
                                f"(attempt {self._backoff.attempt})")
                    time.sleep(delay)
                else:
                    delay = self._backoff.next_delay()
                    logger.warning(f"Unexpected error. Retry in {delay:.1f}s")
                    time.sleep(delay)

            except Exception as e:
                delay = self._backoff.next_delay()
                logger.error(f"Error: {e}. Retry in {delay:.1f}s")
                time.sleep(delay)

            finally:
                if channel:
                    channel.close()

    def stop(self):
        self._running = False


if __name__ == '__main__':
    import sys
    sys.path.insert(0, '../proto')
    import cplane_pb2, cplane_pb2_grpc

    client = DPlaneClient(
        server_address='localhost:50051',
        client_id='dplane-tenant-abc-001',
        tenant_id='tenant-abc',
    )
    try:
        client.run()
    except KeyboardInterrupt:
        client.stop()

📋 Production Checklist

Before deploying a gRPC service to production, verify these items:

Security

  • ☐ TLS enabled (not insecure_channel)
  • ☐ JWT or mTLS authentication
  • ☐ Token rotation implemented
  • ☐ Rate limiting on RegisterClient
  • ☐ gRPC reflection disabled in prod

Reliability

  • ☐ Exponential backoff reconnection
  • ☐ Heartbeat mechanism
  • ☐ Deadlines on unary RPCs
  • ☐ context.is_active() checks in loops
  • ☐ Graceful shutdown (server.stop(grace=5))

Observability

  • ☐ gRPC health check endpoint
  • ☐ Prometheus metrics exported
  • ☐ Structured logging with trace IDs
  • ☐ OpenTelemetry tracing wired
  • ☐ Kubernetes liveness/readiness probes

Protocol

  • ☐ Field numbers never changed
  • ☐ Removed fields are reserved
  • ☐ Proto files versioned in git
  • ☐ Generated stubs committed
  • ☐ Error codes used correctly

✅
Guidance for the Final Challenge
For step 4 (admin push), add a simple FastAPI endpoint to server.py that calls push_api_command(client_id, path) from Lesson 5. This admin HTTP endpoint lets you trigger commands without a separate gRPC tool.

For steps 6-7 (reconnection test), watch the dplane logs — you should see "Server unavailable. Reconnect in 1.0s (attempt 1)", then "...2.0s (attempt 2)", etc. After cplane restarts and becomes healthy, the dplanes reconnect and reset their backoff counters.

If you get UNAVAILABLE errors immediately even though the server is running, check that the port is not firewalled and the TLS cert CN matches the address you're connecting to.
