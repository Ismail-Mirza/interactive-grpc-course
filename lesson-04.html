Unary RPC

Build your first real gRPC server and client in Python. Implement the RegisterClient RPC — the handshake that every dplane performs when connecting to the cplane.

⏱ 60 min 🐍 Python gRPC 📁 Creates: server.py + client.py

🏗️ The Three Core Concepts

To build a gRPC service you need to understand three things:

🔗
Channel
The client's connection to the server. Holds the address, credentials, and connection pool. Think of it as a "connection factory".
📟
Stub
The client-side proxy. Created from a channel. Has one method per RPC. Calling stub.RegisterClient() sends a gRPC request.
🖥️
Servicer
The server-side implementation. You subclass the generated Servicer class and override one method per RPC with your logic.

Unary RPC Request Flow

dplane Client
stub.RegisterClient(req)
→ protobuf bytes
over HTTP/2
cplane Server
RegisterClient(req, ctx)
← RegisterResponse
dplane Client
response.session_id

🖥️ Building the Server (Servicer)

The server extends the generated ControlPlaneServicer and overrides each RPC method. The context parameter gives you metadata, peer info, and the ability to abort with error codes.

pythongrpc-demo/cplane/server.py
import grpc
import uuid
import logging
from concurrent import futures
from datetime import datetime

import sys
sys.path.insert(0, '../proto')
import cplane_pb2
import cplane_pb2_grpc

logging.basicConfig(level=logging.INFO, format='%(asctime)s %(levelname)s %(message)s')
logger = logging.getLogger(__name__)

# In-memory store of connected clients
connected_clients = {}


class ControlPlaneServicer(cplane_pb2_grpc.ControlPlaneServicer):
    """Implements the ControlPlane gRPC service."""

    def RegisterClient(self, request, context):
        """Unary RPC: dplane registers with cplane."""
        logger.info(f"RegisterClient called: client_id={request.client_id} tenant={request.tenant_id}")

        # Validate required fields
        if not request.client_id or not request.tenant_id:
            context.abort(
                grpc.StatusCode.INVALID_ARGUMENT,
                "client_id and tenant_id are required"
            )

        # Simple auth check (real auth in Lesson 8)
        if not request.auth_token:
            context.abort(grpc.StatusCode.UNAUTHENTICATED, "auth_token is required")

        # Register the client
        session_id = str(uuid.uuid4())
        connected_clients[request.client_id] = {
            "tenant_id": request.tenant_id,
            "session_id": session_id,
            "registered_at": datetime.utcnow().isoformat(),
            "version": request.version,
        }

        logger.info(f"Client registered: {request.client_id} → session={session_id}")
        return cplane_pb2.RegisterResponse(
            success=True,
            session_id=session_id,
            message=f"Welcome, {request.client_id}!"
        )

    def GetClientStatus(self, request, context):
        """Unary RPC: get current status of a connected client."""
        client = connected_clients.get(request.client_id)
        if not client:
            context.abort(grpc.StatusCode.NOT_FOUND, f"Client {request.client_id} not found")

        return cplane_pb2.GetStatusResponse(
            client_id=request.client_id,
            status=cplane_pb2.ACTIVE,
        )

    def StreamMessages(self, request_iterator, context):
        """Bidirectional streaming RPC — implemented in Lesson 5."""
        context.abort(grpc.StatusCode.UNIMPLEMENTED, "StreamMessages will be added in Lesson 5")


def serve(port=50051):
    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
    cplane_pb2_grpc.add_ControlPlaneServicer_to_server(ControlPlaneServicer(), server)
    server.add_insecure_port(f'[::]:{port}')
    server.start()
    logger.info(f"cplane gRPC server started on port {port}")
    server.wait_for_termination()


if __name__ == '__main__':
    serve()

📟 Building the Client (Stub)

The client creates a channel to the server's address, then creates a stub from that channel. Each RPC is just a method call on the stub.

pythongrpc-demo/dplane/client.py
import grpc
import logging
import sys

sys.path.insert(0, '../proto')
import cplane_pb2
import cplane_pb2_grpc

logging.basicConfig(level=logging.INFO, format='%(asctime)s %(levelname)s %(message)s')
logger = logging.getLogger(__name__)


class DPlaneClient:
    def __init__(self, server_address='localhost:50051'):
        self.server_address = server_address
        self.channel = None
        self.stub = None
        self.session_id = None

    def connect(self):
        """Create the gRPC channel and stub."""
        # insecure_channel = no TLS (development only, see Lesson 8 for TLS)
        self.channel = grpc.insecure_channel(self.server_address)
        self.stub = cplane_pb2_grpc.ControlPlaneStub(self.channel)
        logger.info(f"Channel created to {self.server_address}")

    def register(self, client_id, tenant_id, auth_token, version="1.0.0"):
        """Call RegisterClient RPC."""
        request = cplane_pb2.RegisterRequest(
            client_id=client_id,
            tenant_id=tenant_id,
            auth_token=auth_token,
            version=version,
        )
        request.capabilities["streaming"] = "true"
        request.capabilities["version"] = version

        try:
            response = self.stub.RegisterClient(request)
            if response.success:
                self.session_id = response.session_id
                logger.info(f"Registered! session_id={self.session_id}")
            return response
        except grpc.RpcError as e:
            logger.error(f"RegisterClient failed: code={e.code()} details={e.details()}")
            raise

    def get_status(self, client_id):
        """Call GetClientStatus RPC."""
        request = cplane_pb2.GetStatusRequest(
            client_id=client_id,
            session_id=self.session_id or "",
        )
        try:
            return self.stub.GetClientStatus(request)
        except grpc.RpcError as e:
            logger.error(f"GetClientStatus failed: {e.code()} {e.details()}")
            raise

    def close(self):
        if self.channel:
            self.channel.close()
            logger.info("Channel closed")


def main():
    client = DPlaneClient('localhost:50051')
    client.connect()

    # Register
    resp = client.register(
        client_id="dplane-tenant-abc-001",
        tenant_id="tenant-abc",
        auth_token="secret-token-123",
        version="2.1.0"
    )
    print(f"Registration: success={resp.success}, session={resp.session_id}")

    # Get status
    status = client.get_status("dplane-tenant-abc-001")
    print(f"Status: client_id={status.client_id}, status={status.status}")

    client.close()


if __name__ == '__main__':
    main()

📊 gRPC Status Codes

gRPC uses its own status codes (not HTTP status codes). These are used in context.abort(code, message) on the server and caught as grpc.RpcError on the client:

grpc.StatusCode.OK
OK
Success
INVALID_ARGUMENT
Invalid Argument
Bad request data
NOT_FOUND
Not Found
Resource doesn't exist
UNAUTHENTICATED
Unauthenticated
Missing/invalid credentials
PERMISSION_DENIED

Authenticated but not authorized

Deadline Exceeded

def DeregisterClient(self, request, context):
    client = connected_clients.get(request.client_id)
    if not client:
        context.abort(grpc.StatusCode.NOT_FOUND,
                      f"Client {request.client_id} not found")

    if client["session_id"] != request.session_id:
        context.abort(grpc.StatusCode.PERMISSION_DENIED,
                      "Invalid session_id")

    del connected_clients[request.client_id]
    logger.info(f"Client deregistered: {request.client_id}")
    return cplane_pb2.DeregisterResponse(
        success=True,
        message=f"Client {request.client_id} deregistered"
    )
