📋 gRPC Metadata — Like HTTP Headers

Metadata is key-value pairs attached to a gRPC call — similar to HTTP headers. The client sends metadata with the request; the server reads it from context.invocation_metadata(). The server can also send metadata back in the response trailing metadata.

pythonClient: sending metadata
import grpc
import uuid

# Option 1: Pass metadata directly to the stub call
response = stub.RegisterClient(
    request,
    metadata=[
        ('x-trace-id', str(uuid.uuid4())),
        ('x-tenant-id', 'tenant-abc'),
        ('x-client-version', '2.1.0'),
        ('authorization', 'Bearer eyJhb...'),
    ]
)

# Option 2: Build reusable metadata as a tuple list
def build_metadata(trace_id=None, auth_token=None):
    meta = []
    if trace_id:
        meta.append(('x-trace-id', trace_id))
    if auth_token:
        meta.append(('authorization', f'Bearer {auth_token}'))
    return meta
pythonServer: reading metadata
def RegisterClient(self, request, context):
    # Read metadata from the incoming request
    meta = dict(context.invocation_metadata())
    trace_id = meta.get('x-trace-id', 'unknown')
    auth_token = meta.get('authorization', '').replace('Bearer ', '')

    logger.info(f"RegisterClient trace_id={trace_id}")

    # Send trailing metadata back to client
    context.set_trailing_metadata([
        ('x-server-version', '1.0.0'),
        ('x-trace-id', trace_id),  # echo back for correlation
    ])

    return cplane_pb2.RegisterResponse(success=True, session_id="abc")
💡
Metadata key naming convention
Metadata keys are lowercase. Binary data uses keys ending in -bin (e.g. auth-token-bin), which are base64-encoded automatically. In FlowGenX's dplane client, auth tokens and trace IDs are sent as metadata on every call.

🔌 Interceptors — gRPC Middleware

Interceptors run before/after every RPC call. They're perfect for: logging, auth, tracing, rate limiting, and metrics. gRPC Python supports both client and server interceptors.

Server Interceptor

pythongrpc-demo/cplane/interceptors.py
import grpc
import time
import uuid
import logging

logger = logging.getLogger(__name__)


class LoggingInterceptor(grpc.ServerInterceptor):
    """Logs every RPC call with duration and trace ID."""

    def intercept_service(self, continuation, handler_call_details):
        method = handler_call_details.method  # e.g. "/cplane.ControlPlane/RegisterClient"
        start = time.monotonic()

        # Get trace ID from metadata
        meta = dict(handler_call_details.invocation_metadata)
        trace_id = meta.get('x-trace-id', str(uuid.uuid4()))

        logger.info(f"→ {method} trace={trace_id}")

        # Call the actual handler
        handler = continuation(handler_call_details)

        # Wrap the handler to measure duration
        if handler and handler.unary_unary:
            original = handler.unary_unary
            def timed_handler(request, context):
                try:
                    result = original(request, context)
                    duration = (time.monotonic() - start) * 1000
                    logger.info(f"← {method} OK {duration:.1f}ms trace={trace_id}")
                    return result
                except Exception as e:
                    logger.error(f"← {method} ERROR {e} trace={trace_id}")
                    raise
            return handler._replace(unary_unary=timed_handler)

        return handler


class AuthInterceptor(grpc.ServerInterceptor):
    """Validates that all calls (except Register) have a valid session."""

    EXEMPT_METHODS = {'/cplane.ControlPlane/RegisterClient'}

    def intercept_service(self, continuation, handler_call_details):
        method = handler_call_details.method
        if method in self.EXEMPT_METHODS:
            return continuation(handler_call_details)

        meta = dict(handler_call_details.invocation_metadata)
        session_id = meta.get('x-session-id')

        if not session_id:
            def deny(request, context):
                context.abort(grpc.StatusCode.UNAUTHENTICATED, "x-session-id header required")
            return grpc.unary_unary_rpc_method_handler(deny)

        # Full JWT validation in Lesson 8
        return continuation(handler_call_details)

Register interceptors on the server

pythonserver.py — add interceptors
from interceptors import LoggingInterceptor, AuthInterceptor

def serve(port=50051):
    # Add interceptors — they run in order
    server = grpc.server(
        futures.ThreadPoolExecutor(max_workers=10),
        interceptors=[
            LoggingInterceptor(),
            AuthInterceptor(),
        ]
    )
    cplane_pb2_grpc.add_ControlPlaneServicer_to_server(ControlPlaneServicer(), server)
    server.add_insecure_port(f'[::]:{port}')
    server.start()
    server.wait_for_termination()

Client Interceptor

pythonClient interceptor — auto-attach trace ID
import grpc
import uuid


class TraceInterceptor(grpc.UnaryUnaryClientInterceptor,
                        grpc.UnaryStreamClientInterceptor,
                        grpc.StreamStreamClientInterceptor):
    """Automatically adds x-trace-id to every outgoing RPC."""

    def _add_metadata(self, client_call_details):
        metadata = list(client_call_details.metadata or [])
        metadata.append(('x-trace-id', str(uuid.uuid4())))
        return client_call_details._replace(metadata=metadata)

    def intercept_unary_unary(self, continuation, client_call_details, request):
        return continuation(self._add_metadata(client_call_details), request)

    def intercept_unary_stream(self, continuation, client_call_details, request):
        return continuation(self._add_metadata(client_call_details), request)

    def intercept_stream_stream(self, continuation, client_call_details, request_iterator):
        return continuation(self._add_metadata(client_call_details), request_iterator)


# Use it when creating the channel
channel = grpc.intercept_channel(
    grpc.insecure_channel('localhost:50051'),
    TraceInterceptor()
)
stub = cplane_pb2_grpc.ControlPlaneStub(channel)

⏱️ Deadlines and Timeouts

A deadline is an absolute point in time by which an RPC must complete. A timeout is a duration. gRPC propagates deadlines automatically through the call chain.

pythonSetting timeouts on client calls
import grpc

# timeout= is in seconds
try:
    response = stub.RegisterClient(request, timeout=5.0)
except grpc.RpcError as e:
    if e.code() == grpc.StatusCode.DEADLINE_EXCEEDED:
        print("RegisterClient timed out after 5 seconds")


# Using an absolute deadline (datetime)
from datetime import datetime, timezone, timedelta

deadline = datetime.now(timezone.utc) + timedelta(seconds=10)
response = stub.RegisterClient(
    request,
    timeout=(deadline - datetime.now(timezone.utc)).total_seconds()
)
pythonServer: checking remaining deadline
def RegisterClient(self, request, context):
    # Check how much time we have left
    time_remaining = context.time_remaining()
    if time_remaining is not None and time_remaining < 0.5:
        context.abort(grpc.StatusCode.DEADLINE_EXCEEDED,
                      "Not enough time to process request")

    # Long operation — periodically check if client is still waiting
    for step in range(10):
        if not context.is_active():
            logger.info("Client cancelled request mid-processing")
            return  # clean exit
        process_step(step)

    return cplane_pb2.RegisterResponse(success=True)
⚠️
Don't set timeouts on streaming RPCs
For StreamMessages, don't use timeout= — the stream is meant to stay open indefinitely. Timeouts are for unary RPCs like RegisterClient and GetClientStatus where you expect a quick response.

💪 Exercise

Build a rate-limiting server interceptor.

Create a RateLimitInterceptor:

1. Track how many times each client_id (from metadata x-client-id) calls RegisterClient per minute

2. If a client calls more than 5 times per minute, abort with RESOURCE_EXHAUSTED and message "Rate limit exceeded: max 5 registrations per minute"

3. Add RetryInfo to the error details with a 60-second retry delay

Hint: use a dict with (client_id, minute_bucket) as key

pythonSolution
from collections import defaultdict
from grpc_status import rpc_status
from google.rpc import status_pb2, code_pb2, error_details_pb2
from google.protobuf.duration_pb2 import Duration

class RateLimitInterceptor(grpc.ServerInterceptor):
    MAX_CALLS = 5
    _counts = defaultdict(int)

    def _bucket(self):
        return int(time.time() // 60)  # minute bucket

    def intercept_service(self, continuation, handler_call_details):
        meta = dict(handler_call_details.invocation_metadata)
        client_id = meta.get('x-client-id', 'unknown')
        key = (client_id, self._bucket())
        self._counts[key] += 1

        if self._counts[key] > self.MAX_CALLS:
            retry_info = error_details_pb2.RetryInfo(
                retry_delay=Duration(seconds=60)
            )
            rich = status_pb2.Status(
                code=code_pb2.RESOURCE_EXHAUSTED,
                message="Rate limit exceeded: max 5 per minute"
            )
            rich.details.add().Pack(retry_info)

            def deny(req, ctx):
                ctx.abort_with_status(rpc_status.to_status(rich))
            return grpc.unary_unary_rpc_method_handler(deny)

        return continuation(handler_call_details)
