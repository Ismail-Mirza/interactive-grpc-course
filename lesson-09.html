💓 gRPC Health Checking Protocol

gRPC has a standard health check protocol (grpc.health.v1) that Kubernetes liveness/readiness probes, load balancers, and service meshes understand natively.

bashInstall health check package
pip install grpcio-health-checking
pythonserver.py — add health checking
from grpc_health.v1 import health, health_pb2, health_pb2_grpc
from grpc_health.v1.health import HealthServicer

def serve(port=50051):
    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))

    # Add your service
    cplane_pb2_grpc.add_ControlPlaneServicer_to_server(ControlPlaneServicer(), server)

    # Add health checking service
    health_servicer = HealthServicer()
    health_pb2_grpc.add_HealthServicer_to_server(health_servicer, server)

    # Set initial health status
    health_servicer.set(
        'cplane.ControlPlane',
        health_pb2.HealthCheckResponse.SERVING
    )
    health_servicer.set(
        '',  # empty string = overall server health
        health_pb2.HealthCheckResponse.SERVING
    )

    server.add_insecure_port(f'[::]:{port}')
    server.start()
    logger.info(f"Server + health check on port {port}")

    # Update health status dynamically based on state
    def update_health():
        while True:
            time.sleep(10)
            if len(connected_clients) > 0:
                health_servicer.set('cplane.ControlPlane',
                                    health_pb2.HealthCheckResponse.SERVING)
            # Could set NOT_SERVING if database is down, etc.

    threading.Thread(target=update_health, daemon=True).start()
    server.wait_for_termination()

pythonMetrics interceptor
import time
from metrics import rpc_requests_total, rpc_duration_seconds

class MetricsInterceptor(grpc.ServerInterceptor):
    def intercept_service(self, continuation, handler_call_details):
        method = handler_call_details.method.split('/')[-1]
        handler = continuation(handler_call_details)

        if handler and handler.unary_unary:
            original = handler.unary_unary
            def timed(request, context):
                start = time.monotonic()
                try:
                    result = original(request, context)
                    rpc_requests_total.labels(method=method, status='OK').inc()
                    return result
                except Exception as e:
                    rpc_requests_total.labels(method=method, status='ERROR').inc()
                    raise
                finally:
                    rpc_duration_seconds.labels(method=method).observe(
                        time.monotonic() - start
                    )
            return handler._replace(unary_unary=timed)
        return handler


# In StreamMessages, track active streams:
def StreamMessages(self, request_iterator, context):
    active_streams.inc()
    try:
        # ... streaming logic ...
        pass
    finally:
        active_streams.dec()

pythonSetup OpenTelemetry tracing
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from opentelemetry.instrumentation.grpc import GrpcInstrumentorServer

def setup_tracing(service_name='cplane-grpc-server'):
    provider = TracerProvider()
    exporter = OTLPSpanExporter(endpoint="http://otel-collector:4317")
    provider.add_span_processor(BatchSpanProcessor(exporter))
    trace.set_tracer_provider(provider)

    # Auto-instrument gRPC server (adds spans for every RPC)
    GrpcInstrumentorServer().instrument()

    return trace.get_tracer(service_name)

# Manual spans within your RPC handler:
tracer = setup_tracing()

def RegisterClient(self, request, context):
    with tracer.start_as_current_span("register-client") as span:
        span.set_attribute("client.id", request.client_id)
        span.set_attribute("tenant.id", request.tenant_id)
        # ... your logic ...
        span.set_attribute("session.id", session_id)
        return response

import pytest
import grpc
from unittest.mock import MagicMock, patch
import sys
sys.path.insert(0, '../proto')
sys.path.insert(0, '../cplane')

import cplane_pb2
from server import ControlPlaneServicer

@pytest.fixture
def servicer():
    return ControlPlaneServicer()

@pytest.fixture
def mock_context():
    ctx = MagicMock()
    ctx.invocation_metadata.return_value = [
        ('x-trace-id', 'test-trace-123')
    ]
    ctx.is_active.return_value = True
    return ctx


class TestRegisterClient:
    def test_success(self, servicer, mock_context):
        request = cplane_pb2.RegisterRequest(
            client_id="test-dplane-1",
            tenant_id="tenant-x",
            auth_token="valid-token",
            version="1.0.0"
        )
        response = servicer.RegisterClient(request, mock_context)
        assert response.success is True
        assert response.session_id != ""
        assert len(response.session_id) == 36  # UUID format

    def test_missing_client_id(self, servicer, mock_context):
        request = cplane_pb2.RegisterRequest(
            tenant_id="tenant-x",
            auth_token="valid-token"
        )
        servicer.RegisterClient(request, mock_context)
        mock_context.abort.assert_called_once_with(
            grpc.StatusCode.INVALID_ARGUMENT,
            "client_id and tenant_id are required"
        )

    def test_missing_auth_token(self, servicer, mock_context):
        request = cplane_pb2.RegisterRequest(
            client_id="dplane-1",
            tenant_id="tenant-x"
        )
        servicer.RegisterClient(request, mock_context)
        mock_context.abort.assert_called_once_with(
            grpc.StatusCode.UNAUTHENTICATED,
            "auth_token is required"
        )

Integration Testing with a real server

pythontests/test_integration.py
import pytest
import grpc
import threading
import time
from concurrent import futures

@pytest.fixture(scope="module")
def grpc_server():
    """Start a real gRPC server for integration tests."""
    from server import ControlPlaneServicer
    import cplane_pb2_grpc

    server = grpc.server(futures.ThreadPoolExecutor(max_workers=2))
    cplane_pb2_grpc.add_ControlPlaneServicer_to_server(ControlPlaneServicer(), server)
    port = server.add_insecure_port('localhost:0')  # random free port
    server.start()
    yield port
    server.stop(grace=1)

@pytest.fixture
def stub(grpc_server):
    channel = grpc.insecure_channel(f'localhost:{grpc_server}')
    yield cplane_pb2_grpc.ControlPlaneStub(channel)
    channel.close()


def test_full_registration_flow(stub):
    """End-to-end: register → get status."""
    # Register
    resp = stub.RegisterClient(cplane_pb2.RegisterRequest(
        client_id="integration-test-1",
        tenant_id="test-tenant",
        auth_token="test-token",
        version="1.0.0"
    ))
    assert resp.success
    assert resp.session_id

    # Get status
    status = stub.GetClientStatus(cplane_pb2.GetStatusRequest(
        client_id="integration-test-1",
        session_id=resp.session_id
    ))
    assert status.client_id == "integration-test-1"
    assert status.status == cplane_pb2.ACTIVE

🔧 grpcurl — Manual gRPC Testing

grpcurl is like curl for gRPC. Use it to test your service from the command line without writing client code:

bashgrpcurl commands
# List services (requires gRPC reflection — see below)
grpcurl -plaintext localhost:50051 list

# Describe a method
grpcurl -plaintext localhost:50051 describe cplane.ControlPlane.RegisterClient

# Call RegisterClient with JSON payload
grpcurl -plaintext -d '{
  "client_id": "test-dplane-1",
  "tenant_id": "tenant-abc",
  "auth_token": "test-token",
  "version": "1.0.0"
}' localhost:50051 cplane.ControlPlane/RegisterClient

# Call with metadata (auth header)
grpcurl -plaintext \
  -H 'authorization: Bearer eyJhb...' \
  -d '{"client_id":"test-1","session_id":"abc"}' \
  localhost:50051 cplane.ControlPlane/GetClientStatus
pythonEnable gRPC reflection (for grpcurl)
from grpc_reflection.v1alpha import reflection
import cplane_pb2

def serve(port=50051):
    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
    cplane_pb2_grpc.add_ControlPlaneServicer_to_server(ControlPlaneServicer(), server)

    # Enable reflection (allows grpcurl to discover services)
    SERVICE_NAMES = (
        cplane_pb2.DESCRIPTOR.services_by_name['ControlPlane'].full_name,
        reflection.SERVICE_NAME,
    )
    reflection.enable_server_reflection(SERVICE_NAMES, server)

    server.add_insecure_port(f'[::]:{port}')
    server.start()
    server.wait_for_termination()

from metrics import registered_clients
connected_clients[request.client_id] = {...}

# Test:
def test_duplicate_registration(stub):
    req = cplane_pb2.RegisterRequest(
        client_id="dup-test", tenant_id="t", auth_token="tok"
    )
    stub.RegisterClient(req)  # first: OK

    with pytest.raises(grpc.RpcError) as exc:
        stub.RegisterClient(req)  # second: ALREADY_EXISTS
    assert exc.value.code() == grpc.StatusCode.ALREADY_EXISTS
